##### C++

```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if(nums.size() == 0)
            return 0;
        
        int j = 0, answer = 1;
        for(int i=1;i<nums.size();i++){
            if(nums[i] != nums[i-1])
                answer++,
                nums[++j] = nums[i];
        }
        
        return answer;
    }
};
```
