### 73. Set Matrix Zeroes

[Question Link](https://leetcode.com/problems/set-matrix-zeroes/)
[Solution Link](https://leetcode.com/submissions/detail/566914784/)

##### C++
```c++
class Solution {
public:
    void setZeroes(vector<vector<int>>& a) {
        int col0 = 1;
        int n = a.size();
        int m = a[0].size();
        
        for(int i=0;i<n;i++){
            if(a[i][0] == 0) col0 = 0;
            for(int j=1;j<m;j++)
                if(a[i][j] == 0)
                    a[i][0] = a[0][j] = 0;
        }
        
        for(int i=n-1;i>=0;i--){
            for(int j=m-1;j>=1;j--){
                if(a[i][0] == 0 || a[0][j] == 0)
                    a[i][j] = 0;
            }
            if(col0 == 0) a[i][0] = 0;
        }
        
        return;
    }
};
```
